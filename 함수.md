# int 타입인지 검사
result = isinstance(7, int) 
print(result)

>>> True 
# float 타입인지 검사 
result = isinstance(0.001, float) 
print(result)

>>> True

# str 타입인지 검사 
result = isinstance(333, str) 
print(result)

>>> False
